4. Entidad JPA - RutaTransporte.java
IMPORTANTE: La tabla rutas_transporte YA EXISTE. JPA no la crear√°, solo la usar√°.
package
com.edumanage.transporte_service.model;

import
lombok.AllArgsConstructor;
import
lombok.Data;
import
lombok.NoArgsConstructor;
import
javax.persistence.*;
import
java.time.LocalTime;

@Data // Getters, setters, toString
@NoArgsConstructor // Constructor vac√≠o
@AllArgsConstructor // Constructor con todos los campos

@Entity // Marca la clase como entidad JPA
@Table(name = "rutas_transporte") // Vincula con la tabla existente
public class RutaTransporte {

    @Id // Llave primaria
    @GeneratedValue(strategy = GenerationType.IDENTITY) // AUTO_INCREMENT
    private Integer id;

    @Column(name = "codigo_ruta") // La columna en BD se llama "codigo_ruta"
    private String codigoRuta;

    @Column(name = "nombre_ruta")
    private String nombreRuta;

    private String descripcion; // Si coincide, @Column es opcional

    @Column(name = "hora_salida")
    private LocalTime horaSalida; // TIME en BD ‚Üí LocalTime en Java

    @Column(name = "hora_llegada")
    private LocalTime horaLlegada;

    private Boolean activo = true; // Valor por defecto
}
